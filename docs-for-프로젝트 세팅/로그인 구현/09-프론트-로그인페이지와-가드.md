# 9. 프론트 — 로그인 페이지 & 보호 라우트 가드

## 목표
- `/login` 페이지 + 로그인 폼 (회원가입 폼과 동일 패턴)
- `RequireAuth` 래퍼로 보호 라우트 감싸기
- 로그인 완료 후 `next` 쿼리 파라미터로 원래 가려던 페이지로 복귀

## LoginForm

### `src/features/auth/login/LoginForm.tsx`
```tsx
"use client";

import Link from "next/link";
import { useRouter, useSearchParams } from "next/navigation";
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { toast } from "sonner";

import { loginSchema, type LoginFormValues } from "@/shared/lib/validation/auth.schema";
import { authActions } from "@/entities/user/model/authStore";
import { getApiError } from "@/shared/api/errors";

export function LoginForm() {
  const router = useRouter();
  const searchParams = useSearchParams();
  const nextPath = searchParams.get("next") || "/dashboard";

  const {
    register,
    handleSubmit,
    setError,
    formState: { errors, isSubmitting },
  } = useForm<LoginFormValues>({
    resolver: zodResolver(loginSchema),
    mode: "onBlur",
    // 테스트 기본값 금지 — mapo 에서 발견한 실수 반복하지 않음
  });

  const onSubmit = handleSubmit(async (values) => {
    try {
      await authActions.login(values.email, values.password);
      toast.success("로그인되었습니다.");
      router.replace(nextPath);
    } catch (e) {
      const api = getApiError(e);
      if (api?.code === "AUTH_003") {
        // 정보 누출 방지: 어느 필드가 틀렸는지 알리지 않음
        setError("password", { type: "server", message: api.message });
      } else if (api?.code === "AUTH_004") {
        setError("email", { type: "server", message: api.message });
      } else {
        toast.error(api?.message ?? "로그인에 실패했습니다.");
      }
    }
  });

  return (
    <form onSubmit={onSubmit} className="space-y-4" noValidate>
      <Field label="이메일" error={errors.email?.message}>
        <input
          type="email"
          autoComplete="email"
          {...register("email")}
          className="w-full rounded-md border border-input bg-background px-3 py-2 text-sm"
        />
      </Field>

      <Field label="비밀번호" error={errors.password?.message}>
        <input
          type="password"
          autoComplete="current-password"
          {...register("password")}
          className="w-full rounded-md border border-input bg-background px-3 py-2 text-sm"
        />
      </Field>

      <button
        type="submit"
        disabled={isSubmitting}
        className="w-full rounded-md bg-primary text-primary-foreground py-2 text-sm font-medium disabled:opacity-60"
      >
        {isSubmitting ? "로그인 중..." : "로그인"}
      </button>

      <p className="text-center text-sm text-muted-foreground">
        계정이 없으신가요?{" "}
        <Link href="/register" className="underline hover:text-foreground">
          회원가입
        </Link>
      </p>
    </form>
  );
}

function Field({ label, error, children }: {
  label: string; error?: string; children: React.ReactNode;
}) {
  return (
    <label className="block space-y-1">
      <span className="text-sm font-medium">{label}</span>
      {children}
      {error && <span className="text-xs text-destructive">{error}</span>}
    </label>
  );
}
```

## 보호 라우트 가드

### `src/widgets/guards/RequireAuth.tsx`
```tsx
"use client";

import { useEffect } from "react";
import { usePathname, useRouter } from "next/navigation";
import { useAuth } from "@/entities/user/model/authStore";

export function RequireAuth({ children }: { children: React.ReactNode }) {
  const { status } = useAuth();
  const router = useRouter();
  const pathname = usePathname();

  useEffect(() => {
    if (status === "anonymous") {
      const next = encodeURIComponent(pathname);
      router.replace(`/login?next=${next}`);
    }
  }, [status, pathname, router]);

  if (status === "authenticated") return <>{children}</>;
  return null;
}
```

## 홈 페이지 라우팅 정리 (선택)
기존 `src/app/page.tsx` 는 이제 랜딩 or 자동 리다이렉트:
```tsx
"use client";
import { useEffect } from "react";
import { useRouter } from "next/navigation";
import { useAuth } from "@/entities/user/model/authStore";

export default function Home() {
  const { status } = useAuth();
  const router = useRouter();
  useEffect(() => {
    if (status === "authenticated") router.replace("/dashboard");
    else if (status === "anonymous") router.replace("/login");
  }, [status, router]);
  return null;
}
```

## 설계 결정
| 결정 | 이유 |
| --- | --- |
| 가드는 **클라이언트 사이드** | Next.js App Router 에서 SSR 보호는 middleware 로 가능하나, JWT localStorage 전제라 클라이언트 가드가 맞음 |
| 가드는 페이지별 래퍼 | layout 에 걸면 `/login` 자체도 가드에 걸림 → 무한 루프 |
| `next` 쿼리 파라미터 | 원래 가려던 페이지로 복귀. 오픈 리다이렉트 방지 위해 **상대 경로만 허용** (검증 추가 권장) |

### 오픈 리다이렉트 방지 (추천 보강)
```ts
const safeNext = nextPath.startsWith("/") && !nextPath.startsWith("//") ? nextPath : "/dashboard";
router.replace(safeNext);
```
